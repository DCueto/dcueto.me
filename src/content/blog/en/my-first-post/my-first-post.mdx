export async function getStaticPaths() {
  const blogPosts = await getCollection("blog") as CollectionEntry<'blog'>[];

  const paths = blogPosts.map((blogPost) => {
    const url = `${blogPost.slug}`;
    const locale = blogPost.data.language;

    return {
      params: {
        locale,
        path: url,
      },
      props: {
        blogPost
      }
    }
  });

  return paths;
}
